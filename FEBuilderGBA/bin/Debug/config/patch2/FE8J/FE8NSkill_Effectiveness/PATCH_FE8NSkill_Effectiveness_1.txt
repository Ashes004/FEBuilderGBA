NAME=FE8NSkill 魔法特効(SkillID:0x17 魔殺し)の対象クラスの指定

IF:$FGREP4 main.dmp=0x00 0xB5 0x30 0xB4

//STRUCTパッチとして実装します
TYPE=STRUCT
TAG=#ENGINE

//データへのポインタ
POINTER=$FGREP4END+16 main.dmp

//データサイズ(10進数)
DATASIZE=1

//データ個数 00 が出てくるまで
DATACOUNT=$GREP1 0x00

//リストに表示されるのはクラス名です
LISTNAME=$B0:CLASS

//データを定義します
B0:CLASS=CLASS
